function get_hand_text($text){
if($text){
preg_match_all("", $text, $match);
if($match[1]){
foreach($match[1] as $val){
$text = str_replace($val, htmlspecialchars(trim($val), ENT_QUOTES), $text);
}
}
return $text;
}
return false;
}
